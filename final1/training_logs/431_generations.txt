\[ \text{Profit per customer} = \text{Revenue per book} - \text{Cost per book} \]

\[ 120 = (\text{Revenue per book} \times 2) - (\text{Cost per book} \times 2) \]

\[ \text{Revenue per customer} = 20 \text{ (profit per book)} + 5 \text{ (cost per book)} = 25 \text{ dollars} \]

\[ \text{Number of customers} = \frac{120}{25} = 4.8 \]
